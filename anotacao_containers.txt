mysql
sudo docker pull mysql/mysql-server:latest

docker run  --publish 6603:3306 --network mysql-network  --detach --name=mysql-docker -e MYSQL_ROOT_PASSWORD=adm -e MY_DATABASE=starbucks  -e MY_USER=root --volume=/root/docker/mysql-standalone/conf.d:/etc/mysql/conf.d mysql

sudo docker exec -it mysql-docker bash

mysql -uroot -p

create database starbucks

show databases;

springboot

1.
clean install maven
incluir .jar gerado na pasta desejada

2.
criar Dockerfile na raiz do projeto

FROM openjdk:11
COPY starbucks-api-1.0.0.jar /projetos/pessoal/starbucks-api/
ENTRYPOINT ["java","-jar","starbucks-api-1.0.0.jar"]

3.
executar comando na pasta do jar(/projetos/pessoal/starbucks-api)
starbucks -> neste caso é o nome da imagem docker
sudo docker build -t starbucks .
